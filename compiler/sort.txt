void main()
{
	int a[20];
	for (int i = 0; i < 20; i = i + 1)
	{
		cin >> a[i];
	}

	for (int i = 0; i < 20; i = i + 1)
	{
		for (int j = i + 1; j < 20; j = j + 1)
		{
			if (a[i] > a[j])
			{
				int t = a[i];
				a[i] = a[j];
				a[j] = t;	
			}		
		}
	}
	
	for (int i = 0; i < 20; i = i + 1)
	{
		cout << a[i];
		cout << " ";
	}
}
