int ToDec(int num, int notation)
{
	int result = 0;
	int m = 1;
	while (num > 0)
	{
		int rest = num % 10;
		result = result + rest * m;
		num = num / 10;
		m = m * notation;
	}
	
	return result;
}

int ToNotation(int num, int notation)
{
	int result = 0;
	int m = 1;
	while (num > 0)
	{
		int rest = num % notation;
		result = result + rest * m;
		num = num / notation;
		m = m * 10;
	}
	
	return result;
}
